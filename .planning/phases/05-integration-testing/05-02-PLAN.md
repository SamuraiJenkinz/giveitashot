    2. Create a helper function `_save_and_open_preview(html_content: str, digest_type: str, logger) -> None` in main.py (above the main() function):
       ```
       def _save_and_open_preview(html_content: str, digest_type: str, logger) -> None:
           """Save HTML digest to output/ and open in default browser."""
           output_dir = Path(__file__).parent.parent / "output"
           output_dir.mkdir(parents=True, exist_ok=True)

           file_path = output_dir / f"{digest_type}_digest.html"
           file_path.write_text(html_content, encoding="utf-8")
           logger.info(f"Digest saved: {file_path.absolute()}")

           # Auto-open in browser
           file_url = file_path.absolute().as_uri()
           try:
               webbrowser.open_new_tab(file_url)
               logger.info(f"Opened in browser: {file_url}")
           except Exception as e:
               logger.warning(f"Could not auto-open browser: {e}")
               logger.info(f"Please manually open: {file_path.absolute()}")
       ```

       NOTE: Use `Path.as_uri()` for cross-platform file:// URL generation (this is the correct Python method). If `as_uri()` is not available in Python 3.10 (it was added in 3.12), fall back to constructing the URL manually: `f"file:///{file_path.absolute().as_posix()}"`.
